# Verified on Python 3.13 (Windows). Every version below is one the acceptance
# checks (scripts/check_task_a|b|c.py) were actually run against — not a version
# picked from a changelog.
#
# Install with:  python -m pip install -r requirements.txt
# NOT with:      pip install -r requirements.txt
# On this machine `pip` belongs to the Microsoft Store Python while `python` is
# the python.org build; `python -m pip` guarantees they are the same interpreter.

# --- data layer (Task A)
pandas==3.0.1
numpy==2.2.6

# --- LLM integration (Task B). The OpenAI SDK talks to LM Studio's
# OpenAI-compatible endpoint; no OpenAI account or key is involved.
openai==2.45.0
python-dotenv==1.2.2

# --- dashboard and charts (Task C)
streamlit==1.59.1
# streamlit 1.59 imports DEFAULT_EXCLUDED_CONTENT_TYPES from starlette's gzip
# middleware, which older starlette releases do not export. Pip will resolve an
# older starlette on a fresh install and streamlit then fails at import with
# `ImportError: cannot import name 'DEFAULT_EXCLUDED_CONTENT_TYPES'`. Pinning it
# forward is the fix.
starlette==1.3.1
plotly==6.9.0
# kaleido renders Plotly figures to PNG/SVG for the exported reports.
# plotly 6.x requires the kaleido 1.x API; 0.2.x will not work here.
kaleido==1.3.0
# kaleido 1.x rasterises by driving a headless Chrome, which Streamlit Cloud has
# no browser for and cannot install one (apt's chromium pulls Debian's own
# libpython and segfaults the container - see the packages.txt history).
# matplotlib is the browserless fallback: src/viz/browserless.py redraws the
# figure with Agg so the deployed reports keep their chart. Locally, where
# Chrome exists, kaleido still does the work and this is never reached.
matplotlib==3.11.0

# --- export (Task C4)
python-docx==1.2.0
reportlab==5.0.0

# --- advanced features (Task D): Isolation Forest for anomaly detection
scikit-learn==1.8.0
# The agent's get_holidays tool calls date.nager.at for the public-holiday
# calendar the dataset does not contain. Streamlit already installs requests, but
# src/data/external.py imports it directly, so it is pinned here rather than
# relied on as somebody else's transitive dependency.
requests==2.34.2
